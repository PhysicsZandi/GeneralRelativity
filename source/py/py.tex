import sympy as sp
import numpy as np
from pprint import pprint

sp.init_printing()

G = sp.Symbol('G')
M = sp.Symbol('M')
t = sp.Symbol('t')
r = sp.Symbol('r')
theta = sp.Symbol("theta") 
phi = sp.Symbol('phi') 
x = [t,r,theta,phi]

def metric(g):
    met = sp.zeros(4, 4)
    for i in range(4):
        for j in range(4):
            met[i,j] = sp.simplify(g[i,j])
    return met

def Christoffel(g):
    g_inv = g.inv()
    c = [sp.zeros(4, 4),  sp.zeros(4, 4), sp.zeros(4, 4), sp.zeros(4, 4)]
    for i in range(4):
        for j in range(4):
            for k in range(4):
                for l in range(4):
                    c[k][i,j] += 1/2 * g_inv[k,l] * (sp.diff(g[i,l], x[j]) + sp.diff(g[j,l], x[i])-sp.diff(g[i,j], x[l]))
                    c[k][i,j] = sp.simplify(c[k][i,j])
    return c

def Riemann(c):
    riemann1 = [sp.zeros(4, 4),  sp.zeros(4, 4), sp.zeros(4, 4), sp.zeros(4, 4)]
    riemann2 = [sp.zeros(4, 4),  sp.zeros(4, 4), sp.zeros(4, 4), sp.zeros(4, 4)]
    riemann3 = [sp.zeros(4, 4),  sp.zeros(4, 4), sp.zeros(4, 4), sp.zeros(4, 4)]
    riemann4 = [sp.zeros(4, 4),  sp.zeros(4, 4), sp.zeros(4, 4), sp.zeros(4, 4)]
    riemann = [riemann1, riemann2, riemann3, riemann4]
    for i in range(4):
        for j in range(4):
            for k in range(4):
                for l in range(4):
                    for a in range(4):
                        riemann[i][j][k,l] += sp.diff(c[i][j,l], x[k]) - sp.diff(c[i][j,k], x[l]) + c[a][j,l] * c[i][a,k] - c[a][j,k] * c[i][a,l] 
                        riemann[i][j][k,l] = sp.simplify(riemann[i][j][k,l])
    return riemann

def Ricci(c): 
    ricci = sp.zeros(4, 4)
    for i in range(4):
        for j in range(4):
            for k in range(4):
                for a in range(4):
                    ricci[i,j] += sp.diff(c[k][i,j], x[k]) - sp.diff(c[k][i,k], x[j]) + c[a][i,j] * c[k][a,k] - c[a][i,k] * c[k][a,j]
                    ricci[i,j] = sp.simplify(ricci[i,j])
    return ricci

def RicciScalar(g, r):
    scalar = 0
    g_inv=g.inv()
    for i in range(4):
        scalar += g_inv[i,i] * r[i,i]
    scalar = sp.simplify(scalar)
    return scalar

def Einstein(g, r, rsc): 
    einstein = sp.zeros(4, 4)
    for i in range(4):
        for j in range(4):
            einstein[i,j] += r[i,j] + 0.5 * g[i,j] * rsc
            einstein[i,j] = sp.simplify(einstein[i,j])
    return einstein

l = sp.Function('lambda')(r)
nu = sp.Function('nu')(r)
g1 = sp.zeros(4, 4)
g1[0,0] = - sp.exp(nu)
g1[1,1] = sp.exp(l)
g1[2,2] = r**2
g1[3,3] = r**2 * sp.sin(theta)**2

m1 = metric(g1)
c1 = Christoffel(g1)
rr1 = Riemann(c1)
r1 = Ricci(c1)
sc1 = RicciScalar(g1, r1)
e1 = Einstein(g1, r1, sc1)
